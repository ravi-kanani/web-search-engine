<ifmodule mod_rewrite.c>
rewriteengine on
rewritecond %{http_user_agent} !feedburner [nc]
rewritecond %{http_user_agent} !feedvalidator [nc]
rewriterule ^feed/?([_0-9a-z-] )?/?$ http://feeds.feedburner.com/wpbeginner [r=302,nc,l]
</ifmodule>
