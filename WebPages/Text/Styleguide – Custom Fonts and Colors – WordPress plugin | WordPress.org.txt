 extra fonts styleguide currently offers developers a filter for adding additional fonts. you can use it as shown below function my_fonts( $font_list ) {
    $font_list['cormorant infant'] = array(
        'name' => 'cormorant infant',
        'family' => '"cormorant infant", serif',
        'weight' => '400,700',
        'sets' => array( 'latin' ),
    );
    $font_list['poppins'] = array(
        'name' => 'poppins',
        'family' => 'poppins, sans-serif',
        'weight' => '400,700',
        'sets' => array( 'latin' ),
    );
    return $font_list;
}
add_filter( 'styleguide_get_fonts', 'my_fonts' );
